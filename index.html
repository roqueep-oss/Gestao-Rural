    GR.Modules.Tarefas = {
        render: function() {
            var div = document.getElementById('lista-acoes');
            if (!div) return;

            var filtrados = items.filter(function(item) {
                if (GR.State.ui.propriedadeAtiva === 'todas') return true;

            if (!filtrados.length) {
                div.innerHTML = '
📋
Nenhuma ação cadastrada
';
                return;

            html += '
TipoDataResponsávelCustoPropriedadeAções
' + GR.Utils.escapeHtml(t.tipo) + '' + GR.Utils.formatarDataBR(t.data) + '' + GR.Utils.escapeHtml(t.responsavel) + '' + GR.Utils.formatarMoedaBR(t.custo) + '' + GR.Utils.escapeHtml(t.propriedade || '-') + '
';
            div.innerHTML = html;
        },

        salvar: function() {
            var tipo = document.getElementById('tarefa-tipo').value.trim();
            var data = document.getElementById('tarefa-data').value;
            var resp = document.getElementById('tarefa-responsavel').value.trim();
            var custo = GR.Utils.parseMoedaBR(document.getElementById('tarefa-custo').value);
            var prop = document.getElementById('tarefa-propriedade').value;

            if (!tipo || !data || !resp) {
                GR.Toast.error('Campos obrigatórios!');
                return;
            }

            var uid = currentUser.uid;
            var dados = {
                tipo: GR.Utils.escapeHtml(tipo),
                data: data,
                responsavel: GR.Utils.escapeHtml(resp),
                custo: custo || 0,
                propriedade: GR.Utils.escapeHtml(prop),
                dataCriacao: GR.Utils.now()
            };

            db.collection('users').doc(uid).collection('tarefas').add(dados)
                .then(function() {
                    GR.Modal.close('modal-tarefa');
                    GR.Toast.success('Ação salva!');
                    GR.State.adicionarHistorico('criou ação', 'Ações', 'Ação: ' + tipo);
                    GR.UI.refreshCurrentView();
                })
                .catch(function(err) {
                    GR.Toast.error('Erro ao salvar: ' + err.message);
                });
        },

        excluir: function(id) {
            if (!confirm('Excluir esta ação?')) return;
            var uid = currentUser.uid;
            db.collection('users').doc(uid).collection('tarefas').doc(id).delete()
                .then(function() {
                    GR.Toast.success('Excluído!');
                    GR.State.adicionarHistorico('excluiu ação', 'Ações', 'Ação ID: ' + id);
                    GR.UI.refreshCurrentView();
                })
                .catch(function(err) {
                    GR.Toast.error('Erro ao excluir: ' + err.message);
                });
        },

        abrirModal: function(editId) {
            GR.State.ui.tarefaEditando = editId || null;
            document.getElementById('modal-tarefa-title').textContent = editId ? '✏️ Editar Ação' : '📋 Nova Ação';

            document.getElementById('tarefa-tipo').value = '';
            document.getElementById('tarefa-data').value = new Date().toISOString().split('T')[0];
            document.getElementById('tarefa-responsavel').value = '';
            document.getElementById('tarefa-telefone').value = '';
            document.getElementById('tarefa-custo').value = '0,00';
            document.getElementById('parte-info-container').style.display = 'none';
            GR.UI._atualizarSelectsPropriedade();

            if (editId) {
                var item = GR.State.data.tarefas.find(function(t) { return t.id === editId; });
                if (item) {
                    document.getElementById('tarefa-tipo').value = item.tipo || '';
                    document.getElementById('tarefa-data').value = item.data || '';
                    document.getElementById('tarefa-responsavel').value = item.responsavel || '';
                    document.getElementById('tarefa-telefone').value = item.telefone || '';
                    document.getElementById('tarefa-custo').value = GR.Utils.formatarMoedaSemSimbolo(item.custo || 0);
                    document.getElementById('tarefa-propriedade').value = item.propriedade || '';
                    GR.Modules.Tarefas.atualizarInfoParte();
                }
            }

            GR.Modal.open('modal-tarefa');
        },

        atualizarInfoParte: function() {
            var nome = document.getElementById('tarefa-responsavel').value.trim();
            var parte = GR.Utils.getPartePorNome(nome);
            var telField = document.getElementById('tarefa-telefone');
            var infoContainer = document.getElementById('parte-info-container');

            if (parte && telField && infoContainer) {
                telField.value = GR.Utils.formatarTelefone(parte.telefone && parte.telefone.ddd, parte.telefone && parte.telefone.numero);
                document.getElementById('parte-nome-exibicao').textContent = GR.Utils.escapeHtml(parte.nome);
                document.getElementById('parte-cpf-exibicao').textContent = GR.Utils.formatarCPF(parte.cpf);
                document.getElementById('parte-telefone-exibicao').textContent = parte.telefone ? GR.Utils.formatarTelefone(parte.telefone.ddd, parte.telefone.numero) : 'Não informado';
                infoContainer.style.display = 'block';
            } else if (telField && infoContainer) {
                telField.value = '';
                infoContainer.style.display = 'none';
            }
        }
    };

    GR.Modules.Contratos = {
        render: function() {
            var div = document.getElementById('lista-contratos');
            if (!div) return;
            var items = GR.State.data.contratos || [];
            var filtrados = items.filter(function(item) {
                if (GR.State.ui.propriedadeAtiva === 'todas') return true;
                return item.propriedade === GR.State.ui.propriedadeAtiva;
            });

            // ===== CÁLCULO DO SALDO DEVEDOR CORRIGIDO =====
            var totalAtivos = 0;
            var saldoDevedor = 0;
            var totalParcelasPendentes = 0;
            var venc15 = 0, venc30 = 0, venc180 = 0, venc365 = 0;
            var hoje = new Date();
            hoje.setHours(0, 0, 0, 0);

            filtrados.forEach(function(c) {
                if (c.status === 'Ativo') totalAtivos++;

                if (c.parcelas && Array.isArray(c.parcelas)) {
                    c.parcelas.forEach(function(p) {
                        if (p.status === 'Pendente' && p.vencimento) {
                            var valorParcela = parseFloat(p.valor) || 0;
                            saldoDevedor += valorParcela;
                            totalParcelasPendentes++;

                            var venc = new Date(p.vencimento);
                            venc.setHours(0, 0, 0, 0);
                            var diff = Math.ceil((venc - hoje) / (1000 * 60 * 60 * 24));
                            if (diff >= 0 && diff <= 15) venc15++;
                            if (diff >= 0 && diff <= 30) venc30++;
                            if (diff >= 0 && diff <= 180) venc180++;
                            if (diff >= 0 && diff <= 365) venc365++;
                        }
                    });
                }
            });

            // ===== STATS =====
            var statsHtml = '
' +
                '
' + totalAtivos + '
📋 Contratos Ativos
' +
                '
' + GR.Utils.formatarMoedaBR(saldoDevedor) + '
💰 Saldo Devedor
' +
                '
' + totalParcelasPendentes + '
📌 Parcelas Pendentes
' +
                '
' + venc15 + '
⚠️ Vence em 15 dias
' +
                '
' + venc30 + '
📅 Vence em 30 dias
' +
                '
' + venc365 + '
📆 Vence em 365 dias
' +
                '
';

            if (!filtrados.length) {
                div.innerHTML = statsHtml + '
💳
Nenhuma operação de crédito
';
                return;
            }

            var rows = filtrados.map(function(c) {
                var totalPendente = 0;
                if (c.parcelas && Array.isArray(c.parcelas)) {
                    c.parcelas.forEach(function(p) {
                        if (p.status === 'Pendente') {
                            totalPendente += parseFloat(p.valor) || 0;
                        }
                    });
                }

                var statusBadge = c.status === 'Ativo' ? 'Ativo' :
                    c.status === 'Quitado' ? 'Quitado' :
                    'Cancelado';

                return '' +
                    '' + GR.Utils.escapeHtml(c.numero) + '' +
                    '' + GR.Utils.escapeHtml(c.propriedade) + '' +
                    '' + GR.Utils.formatarDataBR(c.data) + '' +
                    '' + GR.Utils.formatarMoedaBR(c.valor) + '' +
                    '' + GR.Utils.formatarMoedaBR(totalPendente) + '' +
                    '' + statusBadge + '' +
                    '' +
                    '' +
                    '' +
                    '' +
                    '' +
                    '';
            }).join('');

            div.innerHTML = statsHtml +
                '
' +
                rows + '
NºPropriedadeDataValorPendenteStatusAções
';
        },

            document.getElementById('contrato-modalidade').value = item.modalidade || '';
            document.getElementById('contrato-mutuario').value = item.mutuario || '';
            document.getElementById('contrato-instituicao').value = item.instituicao || '';
            GR.PDFImport.setContratoId(id);
            GR.Modal.open('modal-contrato');
        },

        excluir: function(id) {
            if (!confirm('Excluir esta operação?')) return;
            var uid = currentUser.uid;
            db.collection('users').doc(uid).collection('contratos').doc(id).delete()
                .then(function() {
                    GR.Toast.success('Excluído!');
                    GR.State.adicionarHistorico('excluiu contrato', 'Crédito', 'Contrato ID: ' + id);
                    GR.UI.refreshCurrentView();
                })
                .catch(function(err) {
                    GR.Toast.error('Erro ao excluir: ' + err.message);
                });
        }
    };

📅

';
                return;

                return new Date(b.vencimento) - new Date(a.vencimento);
            });

            var html = '
';
            sorted.forEach(function(p, idx) {
                var originalIdx = parcelas.indexOf(p);
                var hoje = new Date();
                hoje.setHours(0, 0, 0, 0);
                var venc = new Date(p.vencimento);
                venc.setHours(0, 0, 0, 0);
                var diff = Math.ceil((venc - hoje) / (1000 * 60 * 60 * 24));
                var alertClass = '';
                if (p.status === 'Pendente' && diff >= 0 && diff <= 3) alertClass = 'style="background:#ffebee;"';
                else if (p.status === 'Pendente' && diff > 3 && diff <= 7) alertClass = 'style="background:#fff3e0;"';

                var vencimentoValor = p.vencimento || '';

                html += '' +
                    '' +
                    '' +
                    '' +
                    '' +
                    '' +
                    '';

                    valor: parseFloat(valorPorParcela.toFixed(2)),
                    status: 'Pendente',
                    juros: 0
                });

            parcelas.sort(function(a, b) {
                return new Date(b.vencimento) - new Date(a.vencimento);

        processarPDF: function(file) {
            if (file.size > 10 * 1024 * 1024) {
                GR.Toast.error('Arquivo muito grande (máx 10MB).');
                return;
            }

            var progressBar = document.getElementById('analise-progress-bar');
            var progress = document.getElementById('analise-progress');
            if (progressBar) progressBar.style.display = 'block';
            if (progress) progress.style.width = '10%';

            var reader = new FileReader();
            reader.onload = function(e) {
                var arrayBuffer = e.target.result;
                if (typeof pdfjsLib === 'undefined') {
                    GR.Toast.error('PDF.js não carregado.');
                    if (progressBar) progressBar.style.display = 'none';
                    return;
                }

                try {
                    pdfjsLib.GlobalWorkerOptions.workerSrc = 'https://cdnjs.cloudflare.com/ajax/libs/pdf.js/2.16.105/pdf.worker.min.js';
                } catch (err) {}

                pdfjsLib.getDocument({ data: arrayBuffer }).promise
                    .then(function(pdf) {
                        var textoCompleto = '';
                        var totalPages = pdf.numPages;
                        var promises = [];

                        for (var i = 1; i <= totalPages; i++) {
                            promises.push(pdf.getPage(i).then(function(page) {
                                if (progress) progress.style.width = ((i / totalPages) * 50 + 10) + '%';
                                return page.getTextContent().then(function(textContent) {
                                    if (textContent && textContent.items) {
                                        textoCompleto += textContent.items.map(function(item) { return item.str || ''; }).join(' ') + '\n';
                                    }
                                });
                            }));
                        }

                        return Promise.all(promises).then(function() { return textoCompleto; });
                    })
                    .then(function(texto) {
                        if (progress) progress.style.width = '70%';
                        var tipo = document.getElementById('analise-tipo').value;
                        var dados = GR.Analises._extrairDados(texto, tipo);
                        GR.Analises._mostrarResultados(dados);
                        document.getElementById('analise-resultados').style.display = 'block';
                        if (progress) progress.style.width = '100%';
                        setTimeout(function() { if (progressBar) progressBar.style.display = 'none'; }, 500);
                        GR.Toast.success('Análise processada! ' + Object.keys(dados.elementos).length + ' elementos extraídos.');
                    })
                    .catch(function(err) {
                        console.error('Erro ao processar PDF:', err);
                        GR.Toast.error('Erro ao processar: ' + err.message);
                        if (progressBar) progressBar.style.display = 'none';
                    });
            };
            reader.readAsArrayBuffer(file);
        },

    GR.Backup = {
        _intervalo: null,
        _ultimoBackup: null,

        init: function() {
            var self = this;
            this._ultimoBackup = localStorage.getItem('gr_backup_ultimo') || null;

            var hasData = Object.keys(GR.State.data).some(function(k) {
                return GR.State.data[k] && GR.State.data[k].length > 0;
            });

            if (!hasData) {
                var restaurado = this.restaurarAutomatico();
                if (restaurado) {
                    GR.Toast.warning('⚠️ Dados restaurados do backup automático!');
                }
            }

            this._intervalo = setInterval(function() {
                self._salvarBackup();
                self._atualizarBanner();
            }, 6 * 60 * 60 * 1000);

            window.addEventListener('beforeunload', function() {
                self._salvarBackup();
            });

            this._atualizarBanner();
            this._setupRestore();
        },

        openBackupModal: function() {
            GR.Modal.open('modal-backup');
        },

        _salvarBackup: function() {
            try {
                var bk = {};
                Object.keys(GR.State.data).forEach(function(k) {
                    bk[k] = GR.State.data[k];
                });
                var dataStr = JSON.stringify(bk);
                localStorage.setItem('gr_backup_automatico', dataStr);
                this._ultimoBackup = new Date().toISOString();
                localStorage.setItem('gr_backup_ultimo', this._ultimoBackup);
            } catch (e) {
                console.warn('Erro ao fazer backup:', e);
            }
        },

        restaurarAutomatico: function() {
            try {
                var data = localStorage.getItem('gr_backup_automatico');
                if (!data) return false;
                var bk = JSON.parse(data);
                Object.keys(bk).forEach(function(k) {
                    if (GR.State.data.hasOwnProperty(k)) {
                        GR.State.data[k] = bk[k];
                    }
                });
                this._ultimoBackup = localStorage.getItem('gr_backup_ultimo') || null;
                return true;
            } catch (e) {
                console.warn('Erro ao restaurar backup:', e);
                return false;
            }
        },

        exportar: function() {
            var bk = {};
            Object.keys(GR.State.data).forEach(function(k) {
                bk[k] = GR.State.data[k];
            });
            var blob = new Blob([JSON.stringify(bk)], { type: 'application/json' });
            var a = document.createElement('a');
            a.href = URL.createObjectURL(blob);
            a.download = 'backup_' + new Date().toISOString().split('T')[0] + '.json';
            a.click();
            GR.Modal.close('modal-backup');
            GR.Toast.success('Backup exportado!');
            GR.State.adicionarHistorico('exportou backup', 'Sistema', 'Backup exportado');
        },

        fazerBackupManual: function() {
            this._salvarBackup();
            GR.Toast.success('✅ Backup manual realizado!');
            this._atualizarBanner();
        },

        baixarBackup: function() {
            var bk = {};
            Object.keys(GR.State.data).forEach(function(k) {
                bk[k] = GR.State.data[k];
            });
            var blob = new Blob([JSON.stringify(bk)], { type: 'application/json' });
            var a = document.createElement('a');
            a.href = URL.createObjectURL(blob);
            a.download = 'backup_manual_' + new Date().toISOString().split('T')[0] + '.json';
            a.click();
            GR.Toast.success('⬇️ Backup baixado!');
        },

        _atualizarBanner: function() {
            var banner = document.getElementById('backup-banner');
            var timeEl = document.getElementById('backup-last-time');
            if (!banner || !timeEl) return;

            if (this._ultimoBackup) {
                var data = new Date(this._ultimoBackup);
                timeEl.textContent = data.toLocaleString('pt-BR');
                banner.style.display = 'flex';
            } else {
                banner.style.display = 'flex';
                timeEl.textContent = 'Nunca';
            }
        },

        _setupRestore: function() {
            var input = document.getElementById('backup-file');
            if (input) {
                input.addEventListener('change', function(e) {
                    var file = e.target.files[0];
                    if (!file) return;

                    var reader = new FileReader();
                    reader.onload = function(ev) {
                        try {
                            var data = JSON.parse(ev.target.result);
                            Object.keys(data).forEach(function(k) {
                                if (GR.State.data.hasOwnProperty(k)) {
                                    GR.State.data[k] = data[k];
                                }
                            });
                            GR.Toast.success('✅ Restaurado! Recarregando...');
                            e.target.value = '';
                            setTimeout(function() { location.reload(); }, 1500);
                        } catch (err) {
                            GR.Toast.error('❌ Arquivo inválido!');
                            e.target.value = '';
                        }
                    };
                    reader.readAsText(file);
                });
            }
        }
    };

        processarPDF: function(file) {
            if (file.size > 10 * 1024 * 1024) {
                GR.Toast.error('Arquivo muito grande (máx 10MB).');
                return;
            }

            var self = this;
            var progressBar = document.getElementById('nfe-progress-bar');
            var progress = document.getElementById('nfe-progress');
            var statusEl = document.getElementById('nfe-status');
            var ocrStatus = document.getElementById('ocr-status');
            var ocrText = document.getElementById('ocr-progress-text');

            if (progressBar) progressBar.style.display = 'block';
            if (progress) progress.style.width = '10%';
            if (statusEl) statusEl.textContent = '📄 Processando...';
            var limpar = document.getElementById('nfe-btn-limpar');
            if (limpar) limpar.style.display = 'inline-flex';

            var reader = new FileReader();
            reader.onload = function(e) {
                var arrayBuffer = e.target.result;
                if (typeof pdfjsLib === 'undefined') {
                    GR.Toast.error('PDF.js não carregado.');
                    return;
                }

                pdfjsLib.getDocument({ data: arrayBuffer }).promise
                    .then(function(pdf) {
                        var textoCompleto = '';
                        var totalPages = pdf.numPages;
                        var textoEncontrado = false;
                        var promises = [];

                        for (var i = 1; i <= totalPages; i++) {
                            promises.push(pdf.getPage(i).then(function(page) {
                                if (progress) progress.style.width = ((i / totalPages) * 30 + 10) + '%';
                                return page.getTextContent().then(function(textContent) {
                                    var pageText = textContent.items.map(function(item) { return item.str; }).join(' ');
                                    textoCompleto += pageText + '\n';
                                    var cleanText = pageText.replace(/[^a-zA-Z0-9À-ÿ\s.,;:]/g, '');
                                    if (cleanText.length > 50) textoEncontrado = true;
                                });
                            }));
                        }

                        return Promise.all(promises).then(function() {
                            return { texto: textoCompleto, encontrado: textoEncontrado, pdf: pdf };
                        });
                    })
                    .then(function(result) {
                        var textoCompleto = result.texto;
                        var textoEncontrado = result.encontrado;
                        var pdf = result.pdf;
                        if (progress) progress.style.width = '40%';

                        if (!textoEncontrado || textoCompleto.replace(/[^a-zA-Z0-9À-ÿ\s]/g, '').length < 100) {
                            if (ocrStatus) ocrStatus.className = 'ocr-status processing';
                            if (ocrText) ocrText.textContent = '🔄 OCR na imagem...';

                            var ocrPromises = [];
                            for (var i = 1; i <= pdf.numPages; i++) {
                                ocrPromises.push(pdf.getPage(i).then(function(page) {
                                    if (progress) progress.style.width = (40 + (i / pdf.numPages) * 40) + '%';
                                    var viewport = page.getViewport({ scale: 2 });
                                    var canvas = document.createElement('canvas');
                                    canvas.width = viewport.width;
                                    canvas.height = viewport.height;
                                    var ctx = canvas.getContext('2d');
                                    return page.render({ canvasContext: ctx, viewport: viewport }).promise.then(function() {
                                        var imgData = canvas.toDataURL('image/png');
                                        if (typeof Tesseract !== 'undefined') {
                                            return Tesseract.recognize(imgData, 'por', {
                                                logger: function(m) {
                                                    if (m.status === 'recognizing text' && ocrText) {
                                                        ocrText.textContent = '📖 Página ' + i + '/' + pdf.numPages + '... ' + Math.round(m.progress * 100) + '%';
                                                    }
                                                }
                                            }).then(function(result) { return result.data.text; });
                                        }
                                        return '';
                                    });
                                }));
                            }
                            return Promise.all(ocrPromises).then(function(results) {
                                var ocrTextCompleto = results.join('\n');
                                if (ocrStatus) ocrStatus.className = 'ocr-status done';
                                if (ocrText) ocrText.textContent = '✅ OCR concluído!';
                                setTimeout(function() { if (ocrStatus) ocrStatus.className = 'ocr-status'; }, 3000);
                                return ocrTextCompleto;
                            });
                        } else {
                            if (ocrStatus) ocrStatus.className = 'ocr-status done';
                            if (ocrText) ocrText.textContent = '✅ Texto extraído diretamente';
                            setTimeout(function() { if (ocrStatus) ocrStatus.className = 'ocr-status'; }, 3000);
                            return Promise.resolve(textoCompleto);
                        }
                    })
                    .then(function(textoFinal) {
                        if (progress) progress.style.width = '85%';
                        var dados = self._extrairDadosNFE(textoFinal);
                        self.dados = dados;
                        self._mostrarPreviewNFE(dados);
                        var preview = document.getElementById('nfe-preview');
                        if (preview) preview.style.display = 'block';
                        if (progress) progress.style.width = '100%';
                        if (statusEl) statusEl.textContent = '✅ ' + dados.produtos.length + ' produtos encontrados';
                        setTimeout(function() { if (progressBar) progressBar.style.display = 'none'; }, 500);
                        self._atualizarContagemNFE();
                        if (dados.produtos.length === 0) {
                            GR.Toast.warning('Nenhum produto identificado.');
                        } else {
                            GR.Toast.success('NF-e processada! ' + dados.produtos.length + ' produtos.');
                        }
                    })
                    .catch(function(err) {
                        console.error('Erro NF-e:', err);
                        GR.Toast.error('Erro: ' + err.message);
                        if (progressBar) progressBar.style.display = 'none';
                        if (statusEl) statusEl.textContent = '❌ Erro';
                        if (ocrStatus) ocrStatus.className = 'ocr-status error';
                        if (ocrText) ocrText.textContent = '❌ ' + err.message;
                    });
            };
            reader.readAsArrayBuffer(file);
        },

        _extrairDadosNFE: function(texto) {
            var dados = {
                cnpjVendedor: null,
                razaoSocialVendedor: null,
                cnpjComprador: null,
                razaoSocialComprador: null,
                produtos: []
            };

            var cleanText = texto.replace(/\s+/g, ' ').replace(/(\r\n|\n|\r)/g, ' ');

            var cnpjMatch = cleanText.match(/CNPJ\s*[:.]?\s*([0-9]{2}\.?[0-9]{3}\.?[0-9]{3}\/?[0-9]{4}\-?[0-9]{2})/i);
            if (cnpjMatch) {
                dados.cnpjVendedor = cnpjMatch[1].replace(/[.\/\-]/g, '');
            }

            var rsMatch = cleanText.match(/(?:Nome\/Razão\s*Social|Razão\s*Social)\s*[:.]?\s*([A-Za-zÀ-ÿ0-9\s\/\.\-]+?)(?:\s*(?:Endereço|CNPJ|CPF|$)|\n)/i);
            if (rsMatch) {
                dados.razaoSocialVendedor = rsMatch[1].trim().replace(/\s+/g, ' ');
            }

            var compradorMatch = cleanText.match(/(?:CNPJ|CPF)\s*(?:do\s*)?(?:tomador|comprador|cliente)\s*[:.]?\s*([0-9]{2}\.?[0-9]{3}\.?[0-9]{3}\/?[0-9]{4}\-?[0-9]{2}|[0-9]{3}\.?[0-9]{3}\.?[0-9]{3}\-?[0-9]{2})/i);
            if (compradorMatch) {
                var doc = compradorMatch[1].replace(/[.\/\-]/g, '');
                if (doc.length === 14) dados.cnpjComprador = doc;
                else if (doc.length === 11) dados.cnpjComprador = doc;
            }

            var linhas = texto.split('\n');

                            matches.forEach(function(m) {
                                var parts = m.split(':').map(function(s) { return s.trim(); });

                    resultDiv.style.display = 'block';
                    document.getElementById('scan-data').textContent = JSON.stringify(obj, null, 2);

                _configurarDropZones: function() {
                    document.querySelectorAll('.drop-zone').forEach(function(zone) {
                        var newZone = zone.cloneNode(true);
                        zone.parentNode.replaceChild(newZone, zone);
                        newZone.addEventListener('dragover', function(e) {
                            e.preventDefault();
                            this.style.borderColor = '#4caf50';
                            this.style.background = 'rgba(76,175,80,0.1)';
                        });
                        newZone.addEventListener('dragleave', function() {
                            this.style.borderColor = '#ccc';
                            this.style.background = 'transparent';
                        });
                        newZone.addEventListener('drop', function(e) {
                            e.preventDefault();
                            this.style.borderColor = '#ccc';
                            this.style.background = 'transparent';
                            var texto = e.dataTransfer.getData('text/plain');
                            var campo = this.dataset.campo;
                            if (texto && campo) {
                                GR.PDFImport.mapeamentoVisual._adicionarAssociacao(campo, texto);
                            }
                        });
                    });
                },

                _adicionarAssociacao: function(campo, texto) {
                    var camposSimples = ['data_liberacao', 'valor_liberacao', 'mutuario', 'numero_contrato',
                        'taxa_juros', 'saldo_quitacao', 'data_saldo', 'modalidade', 'instituicao'
                    ];
                    if (camposSimples.indexOf(campo) !== -1) {
                        this.associacoes[campo] = texto;
                        this._atualizarDropZone(campo);
                        GR.Toast.success('Campo "' + campo + '" associado: ' + texto);
                        return;
                    }
                    if (campo === 'parcela_aberta_data') {
                        this.parcelasAbertasData.push(texto);
                        this._atualizarDropZone('parcela_aberta_data');
                        GR.Toast.success('Data de parcela em aberto: ' + texto);
                        return;
                    }
                    if (campo === 'parcela_aberta_valor') {
                        this.parcelasAbertasValor.push(texto);
                        this._atualizarDropZone('parcela_aberta_valor');
                        GR.Toast.success('Valor de parcela em aberto: ' + texto);
                        return;
                    }
                    if (campo === 'parcela_liquidada_data') {
                        this.parcelasLiquidadasData.push(texto);
                        this._atualizarDropZone('parcela_liquidada_data');
                        GR.Toast.success('Data de parcela liquidada: ' + texto);
                        return;
                    }
                    if (campo === 'parcela_liquidada_valor') {
                        this.parcelasLiquidadasValor.push(texto);
                        this._atualizarDropZone('parcela_liquidada_valor');
                        GR.Toast.success('Valor de parcela liquidada: ' + texto);
                        return;
                    }
                },

                _atualizarDropZone: function(campo) {
                    var zone = document.querySelector('.drop-zone[data-campo="' + campo + '"]');
                    if (!zone) return;
                    var list = zone.querySelector('.drop-items');
                    if (!list) return;
                    var items = [];
                    if (['data_liberacao', 'valor_liberacao', 'mutuario', 'numero_contrato',
                            'taxa_juros', 'saldo_quitacao', 'data_saldo', 'modalidade', 'instituicao'
                        ].indexOf(campo) !== -1) {
                        if (this.associacoes[campo]) items = [this.associacoes[campo]];
                    } else if (campo === 'parcela_aberta_data') {
                        items = this.parcelasAbertasData;
                    } else if (campo === 'parcela_aberta_valor') {
                        items = this.parcelasAbertasValor;
                    } else if (campo === 'parcela_liquidada_data') {
                        items = this.parcelasLiquidadasData;
                    } else if (campo === 'parcela_liquidada_valor') {
                        items = this.parcelasLiquidadasValor;
                    }
                    list.innerHTML = '';
                    if (items.length === 0) {
                        list.innerHTML = 'Nenhum';
                        return;
                    }
                    items.forEach(function(item, index) {
                        var div = document.createElement('div');
                        div.className = 'drop-item';
                        div.style.cssText = 'display:inline-flex;align-items:center;gap:2px;background:#e8f5e9;color:#2e7d32;padding:1px 4px;border-radius:4px;font-size:8px;margin:1px;';
                        var input = document.createElement('input');
                        input.type = 'text';
                        input.value = item;
                        input.style.cssText = 'border:none;background:transparent;font-size:8px;color:inherit;width:auto;min-width:20px;padding:0 2px;outline:none;';
                        input.size = Math.max(item.length, 4);
                        input.addEventListener('input', function(e) {
                            GR.PDFImport.mapeamentoVisual._atualizarItem(campo, index, e.target.value);
                        });
                        var remove = document.createElement('span');
                        remove.textContent = ' ×';
                        remove.style.cssText = 'cursor:pointer;font-weight:bold;margin-left:2px;color:#c62828;font-size:10px;line-height:1;';
                        remove.addEventListener('click', function(e) {
                            e.stopPropagation();
                            GR.PDFImport.mapeamentoVisual._removerAssociacao(campo, item);
                        });
                        div.appendChild(input);
                        div.appendChild(remove);
                        list.appendChild(div);
                    });
                },

                _atualizarItem: function(campo, index, novoValor) {
                    if (['data_liberacao', 'valor_liberacao', 'mutuario', 'numero_contrato',
                            'taxa_juros', 'saldo_quitacao', 'data_saldo', 'modalidade', 'instituicao'
                        ].indexOf(campo) !== -1) {
                        this.associacoes[campo] = novoValor;
                    } else if (campo === 'parcela_aberta_data') {
                        if (index < this.parcelasAbertasData.length) this.parcelasAbertasData[index] = novoValor;
                    } else if (campo === 'parcela_aberta_valor') {
                        if (index < this.parcelasAbertasValor.length) this.parcelasAbertasValor[index] = novoValor;
                    } else if (campo === 'parcela_liquidada_data') {
                        if (index < this.parcelasLiquidadasData.length) this.parcelasLiquidadasData[index] = novoValor;
                    } else if (campo === 'parcela_liquidada_valor') {
                        if (index < this.parcelasLiquidadasValor.length) this.parcelasLiquidadasValor[index] = novoValor;
                    }
                },

                _removerAssociacao: function(campo, item) {
                    if (['data_liberacao', 'valor_liberacao', 'mutuario', 'numero_contrato',
                            'taxa_juros', 'saldo_quitacao', 'data_saldo', 'modalidade', 'instituicao'
                        ].indexOf(campo) !== -1) {
                        delete this.associacoes[campo];
                    } else if (campo === 'parcela_aberta_data') {
                        var idx = this.parcelasAbertasData.indexOf(item);
                        if (idx !== -1) this.parcelasAbertasData.splice(idx, 1);
                    } else if (campo === 'parcela_aberta_valor') {
                        var idx2 = this.parcelasAbertasValor.indexOf(item);
                        if (idx2 !== -1) this.parcelasAbertasValor.splice(idx2, 1);
                    } else if (campo === 'parcela_liquidada_data') {
                        var idx3 = this.parcelasLiquidadasData.indexOf(item);
                        if (idx3 !== -1) this.parcelasLiquidadasData.splice(idx3, 1);
                    } else if (campo === 'parcela_liquidada_valor') {
                        var idx4 = this.parcelasLiquidadasValor.indexOf(item);
                        if (idx4 !== -1) this.parcelasLiquidadasValor.splice(idx4, 1);
                    }
                    this._atualizarDropZone(campo);
                    GR.Toast.success('Item removido.');
                },

                aplicar: function() {
                    var resultado = {
                        dataLiberacao: this.associacoes['data_liberacao'] || null,
                        valorLiberacao: GR.Utils.parseMoedaBR(this.associacoes['valor_liberacao'] || '0'),
                        mutuario: this.associacoes['mutuario'] || null,
                        numeroContrato: this.associacoes['numero_contrato'] || null,
                        taxaJuros: parseFloat(this.associacoes['taxa_juros'] || '0'),
                        saldoQuitacao: GR.Utils.parseMoedaBR(this.associacoes['saldo_quitacao'] || '0'),
                        dataSaldo: this.associacoes['data_saldo'] || null,
                        modalidade: this.associacoes['modalidade'] || null,
                        instituicao: this.associacoes['instituicao'] || null,
                        parcelasAbertas: [],
                        parcelasLiquidadas: []
                    };
                    var maxAbertas = Math.max(this.parcelasAbertasData.length, this.parcelasAbertasValor.length);
                    for (var i = 0; i < maxAbertas; i++) {
                        var data = i < this.parcelasAbertasData.length ? this.parcelasAbertasData[i] : '';
                        var valor = i < this.parcelasAbertasValor.length ? GR.Utils.parseMoedaBR(this.parcelasAbertasValor[i]) : 0;
                        if (data && valor > 0) {
                            resultado.parcelasAbertas.push({ numero: i + 1, vencimento: data, valor: valor, juros: 0, status: 'Pendente' });
                        }
                    }
                    var maxLiquidadas = Math.max(this.parcelasLiquidadasData.length, this.parcelasLiquidadasValor.length);
                    for (var j = 0; j < maxLiquidadas; j++) {
                        var data2 = j < this.parcelasLiquidadasData.length ? this.parcelasLiquidadasData[j] : '';
                        var valor2 = j < this.parcelasLiquidadasValor.length ? GR.Utils.parseMoedaBR(this.parcelasLiquidadasValor[j]) : 0;
                        if (data2 && valor2 > 0) {
                            resultado.parcelasLiquidadas.push({ numero: j + 1, vencimento: data2, valor: valor2, status: 'Pago' });
                        }
                    }
                    resultado.parcelasAbertas = GR.Utils.ordenarParcelasPorDataDecrescente(resultado.parcelasAbertas);
                    resultado.parcelasLiquidadas = GR.Utils.ordenarParcelasPorDataDecrescente(resultado.parcelasLiquidadas);
                    resultado.parcelasAbertas.forEach(function(p, i) { p.numero = i + 1; });
                    resultado.parcelasLiquidadas.forEach(function(p, i) { p.numero = i + 1; });
                    GR.PDFImport.dadosExtraidos = Object.assign({}, GR.PDFImport.dadosExtraidos, resultado);
                    GR.PDFImport._mostrarPreview();
                    GR.PDFImport._preencherFormulario(resultado);
                    GR.Modal.close('modal-mapeamento-visual');
                    GR.Toast.success('✅ ' + resultado.parcelasAbertas.length + ' abertas, ' + resultado.parcelasLiquidadas.length + ' liquidadas.');
                },

        GR.PDFImport._converterData = function(dataBR) {
            if (!dataBR) return '';
            var partes = dataBR.split('/');
            if (partes.length === 3) {
                var ano = partes[2].padStart(4, '20');
                return ano + '-' + partes[1].padStart(2, '0') + '-' + partes[0].padStart(2, '0');
            }
            return dataBR;
        };

    GR.Voice = {
        reconhecimento: null,
        ativo: false,

        iniciar: function() {
            if (!('webkitSpeechRecognition' in window) && !('SpeechRecognition' in window)) {
                GR.Toast.warning('Seu navegador não suporta reconhecimento de voz.');
                return;
            }

            if (this.ativo) {
                this.ativo = false;
                if (this.reconhecimento) {
                    try { this.reconhecimento.stop(); } catch (e) {}
                }
                GR.Toast.info('Reconhecimento interrompido.');
                return;
            }

            var SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
            this.reconhecimento = new SpeechRecognition();
            this.reconhecimento.lang = 'pt-BR';
            this.reconhecimento.continuous = false;
            this.reconhecimento.interimResults = false;
            this.ativo = true;

            GR.Toast.info('🎤 Ouvindo...');

            this.reconhecimento.onresult = function(event) {

                GR.Toast.success('✅ Comando: "' + resultado + '"');
                GR.Voice._processarComando(resultado);
                GR.Voice.ativo = false;
            };

            this.reconhecimento.onerror = function(event) {
                if (event.error === 'not-allowed') {
                    GR.Toast.error('Permissão de microfone negada.');
                } else {
                    GR.Toast.warning('Erro no reconhecimento. Tente novamente.');
                }
                GR.Voice.ativo = false;
            };

            this.reconhecimento.onend = function() {
                GR.Voice.ativo = false;
            };

            try {
                this.reconhecimento.start();
            } catch (e) {
                GR.Toast.warning('Erro ao iniciar microfone.');
                this.ativo = false;
            }
        },

        _processarComando: function(texto) {
            texto = texto.toLowerCase().trim();

            var comandos = {
                'nova ação': 'modal-tarefa',
                'criar ação': 'modal-tarefa',
                'novo orçamento': 'modal-orcamento',
                'nova operação': 'modal-contrato',
                'novo crédito': 'modal-contrato',
                'novo insumo': 'insumo',
                'abrir dashboard': 'dashboard',
                'abrir relatório': 'relatorios',
                'novo funcionário': 'funcionario',
                'novo animal': 'animal',
                'novo viveiro': 'viveiro',
                'abrir viveiro': 'viveiro',
                'ajuda': 'ajuda'
            };

            for (var chave in comandos) {
                if (texto.includes(chave)) {
                    var acao = comandos[chave];

                    if (acao === 'ajuda') {
                        GR.Toast.info('📋 Comandos disponíveis: Dashboard, Ações, Orçamentos, Crédito, Insumos, Pecuária, Funcionários, Relatórios, Nova Ação, Novo Insumo, Novo Animal, Novo Contrato');
                        return;
                    }

                    if (acao === 'modal-tarefa' || acao === 'modal-orcamento' || acao === 'modal-contrato') {
                        GR.Modal.open(acao);
                    } else if (acao === 'insumo') {
                        GR.Modules.Insumos.abrirModal();
                    } else if (acao === 'funcionario') {
                        GR.Modules.Funcionarios.abrirModal();
                    } else if (acao === 'animal') {
                        GR.Modules.Pecuaria.abrirModalAnimal();
                    } else if (acao === 'viveiro') {
                        GR.UI.navegar('viveiro');
                    } else {
                        GR.UI.navegar(acao);
                    }
                    return;
                }
            }

            GR.Toast.warning('Comando não reconhecido. Diga "ajuda" para ver os comandos disponíveis.');
        }
    };

    GR.ExportCSV = {
        abrirModal: function() {
            GR.Modal.open('modal-exportar-csv');
        },

        exportar: function() {
            var modulo = document.getElementById('csv-modulo-select').value;
            var dados = GR.State.data[modulo] || [];
            if (!dados.length) {
                GR.Toast.warning('Nenhum dado para exportar!');
                return;
            }

            var cabecalho = [];
            var linhas = [];

            if (modulo === 'insumos') {
                cabecalho = ['Nome', 'Categoria', 'Quantidade', 'Unidade', 'Preço', 'Fornecedor', 'Validade', 'Propriedade'];
                linhas = dados.map(function(i) {
                    return [i.nome, i.categoria, i.quantidade, i.unidade, i.preco, i.fornecedor, i.validade, i.propriedade];
                });
            } else if (modulo === 'contratos') {
                cabecalho = ['Número', 'Propriedade', 'Data', 'Valor', 'Status', 'Mutuário', 'Modalidade'];
                linhas = dados.map(function(c) {
                    return [c.numero, c.propriedade, c.data, c.valor, c.status, c.mutuario, c.modalidade];
                });
            } else if (modulo === 'tarefas') {
                cabecalho = ['Tipo', 'Data', 'Responsável', 'Custo', 'Propriedade'];
                linhas = dados.map(function(t) {
                    return [t.tipo, t.data, t.responsavel, t.custo, t.propriedade];
                });
            } else if (modulo === 'funcionarios') {
                cabecalho = ['Nome', 'CPF', 'Cargo', 'Salário', 'Admissão', 'Status', 'Propriedade'];
                linhas = dados.map(function(f) {
                    return [f.nome, f.cpf, f.cargo, f.salario, f.admissao, f.status, f.propriedade];
                });
            } else if (modulo === 'animais') {
                cabecalho = ['Identificação', 'Nome', 'Espécie', 'Raça', 'Nascimento', 'Sexo', 'Peso', 'Propriedade'];
                linhas = dados.map(function(a) {
                    return [a.identificacao, a.nome, a.especie, a.raca, a.nascimento, a.sexo, a.peso, a.propriedade];
                });
            } else {
                cabecalho = Object.keys(dados[0] || {});
                linhas = dados.map(function(item) {
                    return cabecalho.map(function(key) { return item[key] || ''; });
                });
            }

            var csv = cabecalho.join(',') + '\n';
            linhas.forEach(function(row) {
                csv += row.map(function(val) {
                    if (typeof val === 'string' && (val.includes(',') || val.includes('"') || val.includes('\n'))) {
                        return '"' + val.replace(/"/g, '""') + '"';
                    }
                    return val !== undefined && val !== null ? val : '';
                }).join(',') + '\n';
            });

            var blob = new Blob(['\uFEFF' + csv], { type: 'text/csv;charset=utf-8;' });
            var link = document.createElement('a');
            link.href = URL.createObjectURL(blob);
            link.download = modulo + '_' + new Date().toISOString().split('T')[0] + '.csv';
            link.click();
            GR.Modal.close('modal-exportar-csv');
            GR.Toast.success(modulo + ' exportado com sucesso!');
            GR.State.adicionarHistorico('exportou CSV', modulo, 'Exportou dados de ' + modulo);
        }
    };

    GR.ImportCSV = {
        abrirModal: function() {
            document.getElementById('csv-import-name').textContent = 'Nenhum arquivo';
            document.getElementById('csv-import-input').value = '';
            GR.Modal.open('modal-importar-csv');
        },

        importar: function() {
            var input = document.getElementById('csv-import-input');
            if (!input.files || !input.files.length) {
                GR.Toast.warning('Selecione um arquivo CSV!');
                return;
            }

            var modulo = document.getElementById('csv-import-modulo').value;
            var file = input.files[0];
            var reader = new FileReader();

            reader.onload = function(e) {
                try {
                    var text = e.target.result;
                    var lines = text.split('\n').filter(function(l) { return l.trim(); });
                    if (lines.length < 2) {
                        GR.Toast.error('Arquivo vazio ou inválido!');
                        return;
                    }

                    var cabecalho = lines[0].split(',').map(function(h) { return h.trim().replace(/^"|"$/g, ''); });
                    var dados = [];
                    for (var i = 1; i < lines.length; i++) {
                        var valores = lines[i].split(',').map(function(v) {
                            return v.trim().replace(/^"|"$/g, '');
                        });
                        var obj = {};
                        cabecalho.forEach(function(key, idx) {
                            obj[key] = valores[idx] || '';
                        });
                        dados.push(obj);
                    }

                    if (!dados.length) {
                        GR.Toast.error('Nenhum dado válido encontrado!');
                        return;
                    }

                    var colecao = GR.State.data[modulo];
                    if (!colecao) {
                        GR.Toast.error('Módulo não encontrado!');
                        return;
                    }

                    var uid = currentUser.uid;
                    var importados = 0;

                    dados.forEach(function(item) {
                        var id = GR.Utils.gerarId();
                        item.id = id;
                        db.collection('users').doc(uid).collection(modulo).doc(id).set(item)
                            .then(function() {
                                importados++;
                            })
                            .catch(function(err) {
                                console.warn('Erro ao importar item:', err);
                            });
                    });

                    GR.Modal.close('modal-importar-csv');
                    GR.Toast.success('📥 Importando ' + dados.length + ' itens...');
                    GR.State.adicionarHistorico('importou CSV', modulo, 'Importou ' + dados.length + ' itens para ' + modulo);

                    setTimeout(function() {
                        GR.UI.refreshCurrentView();
                    }, 1000);
                } catch (err) {
                    GR.Toast.error('Erro ao importar: ' + err.message);
                }
            };
            reader.readAsText(file);
        }
    };
